JavaScript: Filter out the specified values from a specified array.

    
        const pull = (arr, ...args) => {
            let argState = Array.isArray(args[0]) ? args[0] : args;
            let pulled = arr.filter((v, i) => !argState.includes(v));
            arr.length = 0;
            pulled.forEach(v => arr.push(v));
            return pulled;
          };
